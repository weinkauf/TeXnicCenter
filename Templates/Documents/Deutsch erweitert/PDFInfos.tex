%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% PDF-Informationen
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%
%% ACHTUNG: Sie benötigen ein Hauptdokument, um diese Datei
%%          benutzen zu können. Verwenden Sie im Hauptdokument
%%          den Befehl "\input{dateiname}", um diese
%%          Datei einzubinden.
%%

\pdfinfo{                               % Zusatzinformationen in PDF-Datei;
                                        % alle Werte sind optional.
    /Author (Ihr Name)
    /CreationDate (D:20011010111111)    % Datum der Erstellung
                                        % (D:JJJJMMTThhmmss)
                                        % JJJJ  Jahr
                                        % MM    Monat
                                        % TT    Tag
                                        % hh    Stunden
                                        % mm    Minuten
                                        % ss    Sekunden
                                        %
                                        % Standard: Das aktuelle Datum
                                        %
    /ModDate (D:20011010111111)         % Datum der letzten Modifikation
    /Creator (TeX && TXC)               % Standard: "TeX"
    /Producer (pdfTeX)                  % Standard: "pdfTeX" + pdftex version
    /Title (Titel Ihres Dokumentes)
    /Subject (Thema Ihres Dokumentes)
    /Keywords (Stichwörter zu Ihrem Dokument)
}
