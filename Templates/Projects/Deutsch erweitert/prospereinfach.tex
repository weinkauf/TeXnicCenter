\begin{slide}{Der Punkt}
\begin{itemize}
		\item Das Aussehen des Punkts kann übrigens mit \verb#\myitem{Nummer der Gliederungsebene}{Zeichen}# definiert werden.
		\item \verb#\myitem{2}{$\circ$}# verändert z.\,B. in der zweiten Gliederungsebene den grünen 3D-Punkt in einen schlichten $\circ$.
		\item Das muss man vor \verb#\begin{document}# definieren.
\end{itemize}
\end{slide}
